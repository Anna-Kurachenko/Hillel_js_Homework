function pow(x, y) {
    let result = 1; // початкове значення змінної результату задано 1

    // задаємо логіку циклу, який виконається y разів, щоб помножити x на результат кожної ітерації
    for (let i = 0; i < y; i++) {
        result *= x; // множимо х на поточний результат ітерації
    }

    return result; // повертаємо кінцевий результат після завершення циклу
}

// виклик функції та виведення результату
console.log(pow(2, 3)); // Очікуваний результат: 8
console.log(pow(3, 4)); // Очікуваний результат: 81
